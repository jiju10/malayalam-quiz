let shuffled = [];

let current = 0;

let score = 0;


function shuffle(){

  shuffled = questions.sort(() => Math.random() - 0.5);

}


function startQuiz(){

  if(!name.value || !age.value){

    alert("വിവരങ്ങൾ പൂരിപ്പിക്കുക");

    return;

  }


  localStorage.setItem("user", JSON.stringify({

    name:name.value,

    age:age.value,

    sex:sex.value

  }));


  attempts++;

  localStorage.setItem("attempts", attempts);


  document.getElementById("userForm").classList.add("hidden");

  document.getElementById("quiz").classList.remove("hidden");


  shuffle();

  loadQ();

}


function loadQ(){

  let q = shuffled[current];

  qText.innerText = (current+1)+". "+q.q;

  options.innerHTML = "";


  q.o.forEach((opt,i)=>{

    let btn = document.createElement("button");

    btn.innerText = opt;

    btn.onclick = ()=>{

      if(i === q.a) score++;

      nextQ();

    };

    options.appendChild(btn);

  });

}


function nextQ(){

  current++;

  if(current < 25){

    loadQ();

  } else {

    showResult();

  }

}


function showResult(){

  quiz.classList.add("hidden");

  result.classList.remove("hidden");


  let percent = (score/25)*100;


  let pass = percent >= 70 ? "വിജയം 🎉" : "പരാജയം";


  result.innerHTML = `

  <h2>ഫലം</h2>

  മാർക്ക്: ${score}/25 <br>

  ശതമാനം: ${percent.toFixed(2)}% <br>

  സ്ഥിതി: ${pass}

  `;

}
